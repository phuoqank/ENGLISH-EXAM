// Hàm khởi chạy đề thi
function startQuiz() {
const name = document.getElementById("student-name").value.trim();
if(!name) {
alert("Em vui lòng nhập họ và tên trước khi làm bài nhé!");
return;
}

document.getElementById("student-info").style.display = "none";
document.getElementById("quiz-container").style.display = "block";
document.getElementById("quiz-title").innerText = quizData.quizTitle;

let html = "";
quizData.questions.forEach((q, index) => {
html += `
<div class="question-box">
<p><b>Câu ${index + 1}:</b> ${q.questionText}</p>
<div class="options-list">
`;
q.options.forEach(opt => {
const optLetter = opt.trim().charAt(0);
html += `
<label class="option-item">
<input type="radio" name="${q.id}" value="${optLetter}"> ${opt}
</label>
`;
});
html += `</div></div>`;
});
document.getElementById("questions-list").innerHTML = html;
}

// Hàm tự động chấm điểm và gửi lên Google Sheets
function submitQuiz() {
const studentName = document.getElementById("student-name").value;
const submitBtn = document.getElementById("submit-btn");

let answeredCount = 0;
quizData.questions.forEach(q => {
if(document.querySelector(`input[name="${q.id}"]:checked`)) answeredCount++;
});

if(answeredCount < quizData.questions.length) {
if(!confirm("Em chưa hoàn thành hết tất cả các câu hỏi. Vẫn muốn nộp bài chứ?")) {
return;
}
}

submitBtn.disabled = true;
submitBtn.innerText = "Đang nộp bài lên hệ thống...";

let correctCount = 0;
let totalQuestions = quizData.questions.length;

quizData.questions.forEach(q => {
const selected = document.querySelector(`input[name="${q.id}"]:checked`);
const answer = selected ? selected.value : "Chưa chọn";

if(answer === q.correctAnswer) {
correctCount++;
}
});

let score = parseFloat(((correctCount / totalQuestions) * 10).toFixed(2));

let comment = "";
if (score >= 9) comment = "Xuất sắc! Em nắm rất chắc kiến thức từ loại.";
else if (score >= 7) comment = "Khá tốt! Phát huy thêm em nhé.";
else if (score >= 5) comment = "Đạt yêu cầu. Cần ôn tập kỹ các quy tắc hậu tố cấu tạo từ.";
else comment = "Chưa đạt. Em cần học lại cách phân biệt Danh/Tính/Động/Trạng từ.";

const payload = {
studentName: studentName,
quizId: quizData.quizId,
correctCount: `${correctCount}/${totalQuestions}`,
score: score,
comment: comment
};

fetch(WEB_APP_URL, {
method: "POST",
mode: "no-cors",
headers: { "Content-Type": "application/json" },
body: JSON.stringify(payload)
})
.then(() => {
document.getElementById("quiz-container").style.display = "none";
const resultBox = document.getElementById("result");
resultBox.style.display = "block";
resultBox.innerHTML = `
<h2 style="color: #28a745; margin-top:0;">NỘP BÀI THÀNH CÔNG!</h2>
<p style="font-size: 16px;"><b>Học sinh:</b> ${studentName}</p>
<p style="font-size: 16px;"><b>Kết quả:</b> Làm đúng ${correctCount}/${totalQuestions} câu</p>
<div style="margin: 20px auto; width: 100px; height: 100px; background: #fff; border-radius: 50%; line-height: 100px; border: 3px solid #28a745;">
<span style="color:red; font-size: 28px; font-weight:bold;">${score}</span>
</div>
<p style="font-size: 16px; font-style: italic;"><b>Nhận xét hệ thống:</b> "${comment}"</p>
<p style="color: #666; font-size: 13px; margin-top: 20px;">✓ Điểm số đã được ghi nhận tự động vào Sổ điểm của Giáo viên.</p>
`;
})
.catch(error => {
console.error("Lỗi:", error);
alert("Mất kết nối mạng tạm thời khi nộp điểm. Hãy thử lại!");

submitBtn.innerText = "Nộp bài & Xem kết quả";
});
